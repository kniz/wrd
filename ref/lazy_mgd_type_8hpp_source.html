13  // when you make an instance of nInt, it will try to make an instance of tbaseObjOrigin<nInt>.
14  // however this will trigger chain of infinite loop. see how it goes.
15  //
16  // tbridger<nInt>::ctor<nInt>() { return new tbridgeFunc<Args...>(); }
17  // tbridger<nInt>'s ctor() will try to make an instance of tbridgeFunc, but,
18  //
19  // tbridgeFunc<Args..>::tbridgeFunc<Args...>():
20  // tbaseBridgeFunc(Marshaling<Args...>::onGetRet());
21  // as you can see, it calls onGetRet() to include returnType to mgdType of tbridgeFunc.
22  // guess what it'll happen if you call onGetRet().
23  //
24  // Marshaling<T, Args...>::onGetRet() { return new T(); }
25  // in this scenario, T is nInt. so, it eventually calls constructor of nInt.

27  //
28  // nInt::nInt(): super(tbaseObjOrigin<nInt>(tbridge<T>::ctor<nInt>())) {}
29  // we goes back to first step. and you can see beautiful unexpected infinite loop.
